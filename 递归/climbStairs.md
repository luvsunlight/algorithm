/**
 * @param {number} n
 * @return {number}
 */
var climbStairs = function(n, map = { 1: 1, 2: 2 }) {
	if (map[n]) return map[n]
	else map[n] = map[n - 1] + map[n - 2]
	return climbStairs(n - 1, map) + climbStairs(n - 2, map)
}
